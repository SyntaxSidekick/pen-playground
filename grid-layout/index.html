Grid Template Areas

Header
Sidebar
Main Content
Footer

HTML

<div class="grid-container template-areas">
    <div class="grid-item header">Header</div>
    <div class="grid-item sidebar">Sidebar</div>
    <div class="grid-item content">Main Content</div>
    <div class="grid-item footer">Footer</div>
</div>

CSS

.template-areas {
    display: grid;
    grid-template-areas: 
        "header header"
        "sidebar content"
        "footer footer";
    grid-template-columns: 200px 1fr;
    gap: 1rem;
}

.header { grid-area: header; }
.sidebar { grid-area: sidebar; }
.content { grid-area: content; }
.footer { grid-area: footer; }

Grid Lines and Spanning

Spans 2 Columns
Item 2
Item 3
Spans 2 Rows
Item 5

HTML

<div class="grid-container spanning-grid">
    <div class="grid-item span-item-1">Spans 2 Columns</div>
    <div class="grid-item span-item-2">Item 2</div>
    <div class="grid-item span-item-3">Item 3</div>
    <div class="grid-item span-item-4">Spans 2 Rows</div>
    <div class="grid-item span-item-5">Item 5</div>
</div>

CSS

.spanning-grid {
    display: grid;
    grid-template-columns: repeat(3, 1fr);
    gap: 1rem;
}

.span-item-1 {
    grid-column: 1 / 3; /* Spans 2 columns */
}

.span-item-4 {
    grid-row: 2 / 4; /* Spans 2 rows */
}

HTML

<div class="grid-container nested-grid">
    <div class="grid-item nested-parent">
        <h4>Nested Grid 1</h4>
        <div class="inner-grid">
            <div class="inner-item">A</div>
            <div class="inner-item">B</div>
            <div class="inner-item">C</div>
            <div class="inner-item">D</div>
        </div>
    </div>
</div>

CSS

.nested-grid {
    grid-template-columns: repeat(2, 1fr);
    gap: 2rem;
}

.inner-grid {
    display: grid;
    grid-template-columns: repeat(2, 1fr);
    gap: 0.5rem;
    margin-top: 1rem;
    
    &.column-grid {
        grid-template-columns: 1fr;
    }
}

Dashboard Layout Example

Navigation
Main Chart
Stats
Widget 1
Widget 2
Widget 3
Footer Info

HTML

<div class="dashboard-grid">
    <div class="dashboard-item nav">Navigation</div>
    <div class="dashboard-item main-chart">Main Chart</div>
    <div class="dashboard-item sidebar-stats">Stats</div>
    <div class="dashboard-item widget-1">Widget 1</div>
    <div class="dashboard-item widget-2">Widget 2</div>
    <div class="dashboard-item widget-3">Widget 3</div>
    <div class="dashboard-item footer-info">Footer Info</div>
</div>

CSS

.dashboard-grid {
    display: grid;
    grid-template-areas: 
        "nav nav nav sidebar"
        "main main main sidebar"
        "widget1 widget2 widget3 sidebar"
        "footer footer footer footer";
    grid-template-columns: repeat(3, 1fr) 200px;
    grid-template-rows: 60px 200px 120px 60px;
}

.nav { grid-area: nav; }
.main-chart { grid-area: main; }
.sidebar-stats { grid-area: sidebar; }
.widget-1 { grid-area: widget1; }
.widget-2 { grid-area: widget2; }
.widget-3 { grid-area: widget3; }
.footer-info { grid-area: footer; }
